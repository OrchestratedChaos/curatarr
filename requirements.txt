# Core Python dependencies for Plex recommendation system (CLI/cron).
#
# Pinned to exact versions so `pip install -r requirements.txt` (no
# --upgrade) is reproducible. Bump deliberately, one PR at a time.
#
# This is the *core* dependency set only - just what the CLI/cron
# recommendation engine needs (movie/tv/external recommenders). The
# local web UI's own deps (flask, ruamel.yaml) live in
# requirements-ui.txt so that a plain `./run.sh` update for existing
# CLI/cron users never pulls in the (heavier) UI stack. See
# requirements-ui.txt and run-ui.sh/run-ui.ps1.
#
# run.sh/run.ps1 actually install from requirements.lock (this file's
# direct + transitive deps, fully SHA256-hashed, `pip install
# --require-hashes`) for supply-chain integrity beyond version pinning.
# After changing this file, regenerate requirements.lock - see the
# comment at the top of that file for the command.
#
# Python floor: plexapi>=4.18.0 and requests>=2.33.0 both require
# Python 3.10+, so 3.10 is the effective floor for this file regardless
# of which of the two drives it - see requirements.lock's header for
# the full explanation of why neither can be pinned back down to
# restore Python 3.8/3.9 support without dropping the requests
# CVE-2026-25645 fix.

# Core Plex API client
plexapi==4.18.2

# HTTP requests library. Pinned at 2.34.2 for the CVE-2026-25645 fix
# (GHSA-gc5v-m9x4-r6x2, insecure temp file reuse in
# extract_zipped_paths() - a function curatarr itself never calls, but
# the pin is kept as defense in depth). Note this alone already
# requires Python>=3.10 (first-patched 2.33.0 does too) - independent
# of the plexapi floor above.
requests==2.34.2

# YAML configuration parser
pyyaml==6.0.3

# Pure-Python Ed25519/SSHSIG verification for the standalone binaries'
# in-binary self-updater (utils/self_update.py) - verifies
# SHA256SUMS.txt.sig against the pinned release-signing key before
# trusting any downloaded binary's hash. A core (not build-only)
# dependency: this runs at UPDATE time inside the packaged binary
# itself, on the end user's machine, not just at CI build time - see
# utils/self_update.py's module docstring for why this is pure Python
# rather than shelling out to a system `ssh-keygen`.
#
# Pinned at 48.0.1, NOT the newer 49.0.0, because 49.0.0 dropped macOS
# x86_64 wheels entirely (arm64-only per its own CHANGELOG: 'Support
# for x86_64 macOS has been removed') - no x86_64 wheel exists to fuse
# into a universal2 build at all, which is what broke the v2.8.29
# macos-universal release (PyInstaller bundled the arm64-only
# _rust.abi3.so, IncompatibleBinaryArchError). 48.0.1 is the newest
# release that still ships a real prebuilt macosx_10_9_universal2
# wheel on PyPI - no delocate-merge fuse needed for this dep, plain
# pip install picks it up directly - and it already carries the fixed
# OpenSSL build (GHSA-537c-gmf6-5ccf); no known CVE is unpatched
# between 48.0.1 and 49.0.0 (checked pyca/cryptography's GHSA
# advisories directly). Bump only alongside re-verifying PyPI still
# publishes a macOS universal2 wheel for the target version.
cryptography==48.0.1
